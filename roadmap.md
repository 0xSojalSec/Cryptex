# Arguments
-e = encrypt
-d = decrypt
-b = break [default, check for wordlist]
    -w = wordlist
-t = text input [one string]
-f = file input
-o = file output

# Symbols
\> = direct to

# Ciphers
-cc = caesar cipher
-vc = vigenere cipher
-mc = multiplicative cipher
-rc = reverse cipher
-dh = diffie hellman key exchange
-xor = XOR cipher
-r13 = ROT 13
-r47 = ROT 47

# Encodings 
-b64 = base64
-bin = binary
-hex = hexadecimal
-oct = octal
-1337 = 1337 5P34K
-mor = morse code
-pho = phonetic alphabet

# Hashes
-md5 = MD5
-sha256 = SHA256
